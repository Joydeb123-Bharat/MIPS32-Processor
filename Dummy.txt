// Initialization
ADDI R1, R0, 0    // R1 = Base Memory Address (starts at 0)
ADDI R2, R0, 10   // R2 = Data Value to write (starts at 10)
ADDI R3, R0, 5    // R3 = Loop Counter (starts at 5)
ADDI R4, R0, 1    // R4 = Constant 1 (for decrementing the counter)
ADDI R5, R0, 4    // R5 = Constant 4 (for incrementing the memory address)

// Main Loop
LOOP:
SW R2, 0(R1)      // Store the Data Value (R2) into memory at the Address (R1)
ADD R2, R2, R2    // Wait, we don't have a multiply or immediate add for R2 easily without another register. Let's just use ADDI if your assembler supports it.
// Actually, ADDI R2, R2, 10 works perfectly!
ADDI R2, R2, 10   // Increment Data Value by 10
ADD R1, R1, R5    // Increment Memory Address by 4 (Move to next word slot)
SUB R3, R3, R4    // Decrement Loop Counter by 1
BNEQZ R3, LOOP    // If Counter != 0, branch back to LOOP

HLT               // Stop the processor
